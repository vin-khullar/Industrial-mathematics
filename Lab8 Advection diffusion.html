# FTCS method for 1D diffusion eq; stable only till D <0.5
# D=0 gives only addvection, V=0 gives only diffuiosn

D=0.3; L=1.0; Time=0.1; miu=0.5; sigma=0.07;vel=2

NX=100; NT=1000

dx=L/(NX-1); dt=Time/NT

x=linspace(0.0,1.0,NX)
C=zeros(NX)
for i in range(0,NX,1):
    C[i]=(1/sqrt(2*pi*sigma**2))*exp(-((x[i]-miu)**2)/(2*sigma**2))
#C=sin(2*pi*x)

RHS=zeros(NX)

for n in range(0, NT):
    
    for j in range (1, NX-1):
        C[j] += D*dt*(C[j+1]-2*C[j]+C[j-1])/(dx*dx)-vel*dt*(C[j+1]-C[j])/dx
    
    if (n%100 == 0):
        plt.plot(x,C)
plt.axis([-0.1,1.2,-0.1,6])
